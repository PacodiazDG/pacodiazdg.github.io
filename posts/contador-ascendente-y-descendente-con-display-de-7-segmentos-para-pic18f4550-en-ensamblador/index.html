contador ascendente y descendente con Display de 7 Segmentos para PIC18F4550 en Ensamblador

2025-06-07
#assembly  #PIC18F4550 

En este proyecto se va a desarrollar un contador ascendente y descendente utilizando el microcontrolador PIC18F4550 en ensamblador. usando un Display de 7 Segmentos para mostrar los valores del contador en orden ascendente y descendente.

algunos de los pasos se expican en el contador automatico que se encuentra en

https://devnotescode.com/Pages?id=677437501f54d06c6ecb8012



LIST    P = 18F4550
INCLUDE <P18F4550.INC>
;contador de cambio automatico 
;************************************************************

CONFIG  PWRT = ON
CONFIG  BOR = OFF
CONFIG  WDT = OFF

; CODE ******************************************************
    
  
ORG 0x00    ;Iniciar el programa en el registro 0x00
    movlw   0xFF                ; 
    movwf  TRISC
    movlw   0x00        ; 
    movwf  TRISB
; variables
    Counter1  equ 0x20 
    Counter2  equ 0x21 ; 
    
    sa  equ 0x22        
    movlw   0x00

main:
    ;--------------------------------------------------------------------

    ;incf    sa, f   aumenta

    goto main
    return

 ;--------------------------------- Genera un sleep aprox 2 segundos
 Delay2s:
    movlw   D'50'       
    movwf   Counter1    
DelayLoop1:
    movlw   D'200'      
    movwf   Counter2    
DelayLoop2:
    nop                
    nop                            
    decfsz  Counter2, f 
    goto    DelayLoop2  
    decfsz  Counter1, f 
    goto    DelayLoop1  
    return

;********************************************************************************************************
revisarintoverflow:
    nop
    call esmin0 ; no permite que llege a 0
    call esmax15 ; no permite que llege a 15
    return
esmax15:
    nop
    MOVLW 16
    SUBWF sa, W
    BTFSS STATUS, Z
    return ;// esto es igual a !=
    movlw   0x00       ; Cargar el nuevo valor (0x01) en el registro W
    movwf   sa         ; Transferir el valor de W a 'sa'
    return
esmin0:
    movlw   0xFF       
    subwf   sa, W      ; 
    btfss   STATUS, Z  ; esto es igual a != 
    return            ; 
    movlw   15       
    movwf   sa         
    return             
;********************************************************************************************************
    ;incf    sa, f   aumenta
    ;decf    sa, f   decrementa
    acsListener:
    BTFSS PORTC, 0  ; Comprobar si el botn en RC0 est presionado 
    incf    sa, f
    return
    descListener:
    BTFSS PORTC, 1  ; Comprobar si el botn en RC1 est presionado
    decf    sa, f    
    return
    controldebotones:
	call acsListener
	call descListener
